En caso de tener problemas. Usar el siguiente consulta para la creación de la base de datos, el entrar en ella, y crear la tabla desde la consola.

CREATE DATABASE IF NOT EXISTS generador_cv;
USE generador_cv;

CREATE TABLE datos_cv (
  id INT AUTO_INCREMENT PRIMARY KEY,
  version_cv INT NOT NULL,
  nombre VARCHAR(50) NOT NULL,
  apellidos VARCHAR(50) NOT NULL,
  email VARCHAR(150) NOT NULL,
  telefono VARCHAR(50),
  domicilio VARCHAR(150),
  experiencia TEXT,
  formacion TEXT,
  habilidades TEXT,
  idiomas TEXT,
  foto VARCHAR(255),
  fecha_creacion DATETIME DEFAULT CURRENT_TIMESTAMP
);
